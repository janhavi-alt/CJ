/*
 * Program 3(a): Dynamic Initialization
 */
package pract3a;

public class Pract3a {
    public static void main(String[] args) {
        double a = 5.5;
        double b = 2.5;

        double result = Math.sqrt(a * a + b * b); // dynamic initialization

        System.out.println("Value of a = " + a);
        System.out.println("Value of b = " + b);
        System.out.println("Dynamically initialized result = " + result);
    }
}


/*
 * Program 3(b): Area and Perimeter of a Circle
 */
package pract3b;

class Circle {
    double radius;

    void setRadius(double r) {
        radius = r;
    }

    double area() {
        return Math.PI * radius * radius;
    }

    double perimeter() {
        return 2 * Math.PI * radius;
    }
}

public class Pract3b {
    public static void main(String[] args) {
        Circle c = new Circle();
        c.setRadius(5.0);

        System.out.println("Radius = " + c.radius);
        System.out.println("Area = " + c.area());
        System.out.println("Perimeter = " + c.perimeter());
    }
}


/*
 * Program 3(c): Access member variables using constructor
 */
package pract3c;

class Student {
    String name;
    int age;

    Student(String n, int a) {
        name = n;
        age = a;
    }

    void display() {
        System.out.println("Name: " + name);
        System.out.println("Age : " + age);
    }
}

public class Pract3c {
    public static void main(String[] args) {
        Student s = new Student("Janhavi", 23);
        s.display();
    }
}
